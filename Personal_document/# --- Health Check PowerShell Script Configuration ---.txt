# --- Configuration ---
$SQLInstance = "localhost" # or "M903PRD1"
$EmailTo = "Mukesh.chaurasia@brisbane.qld.gov.au"
$EmailFrom = "M903PRD1@brisbane.qld.gov.au"
$SMTPServer = "smtp.bcc.qld.gov.au"
$ThresholdGB = 20 # Alert if disk space is below this

# --- 1. SQL Service Status ---
$Services = Get-CimInstance Win32_Service | Where-Object { $_.Name -like "*SQL*" -or $_.Name -like "*SQLAgent*" } | 
    Select-Object Name, State, Status, StartMode

# --- 2. Disk Space (Server Level) ---
$Disks = Get-CimInstance Win32_LogicalDisk -Filter "DriveType=3" | Select-Object DeviceID, 
    @{N="Size(GB)";E={[math]::Round($_.Size/1GB,2)}}, 
    @{N="Free(GB)";E={[math]::Round($_.FreeSpace/1GB,2)}},
    @{N="Free(%)";E={[math]::Round(($_.FreeSpace/$_.Size)*100,2)}}

# --- 3. Database Info (Growth & Size) ---
$DBQuery = @"
SELECT name AS [DBName], 
    CAST(size*8/1024.0 AS DECIMAL(10,2)) AS [SizeMB],
    state_desc AS [Status],
    recovery_model_desc AS [RecoveryModel]
FROM sys.databases
"@
$DBInfo = Invoke-SqlCmd -ServerInstance $SQLInstance -Query $DBQuery

# --- 4. Backup Status (Last 24 Hours) ---
$BackupQuery = @"
SELECT d.name AS DatabaseName, 
    MAX(b.backup_finish_date) AS LastBackupDate,
    CASE WHEN MAX(b.backup_finish_date) > DATEADD(hh, -24, GETDATE()) THEN 'Current' ELSE 'OUTDATED' END AS [Status]
FROM sys.databases d
LEFT JOIN msdb.dbo.backupset b ON d.name = b.database_name AND b.type = 'D'
WHERE d.name <> 'tempdb'
GROUP BY d.name
"@
$BackupStatus = Invoke-SqlCmd -ServerInstance $SQLInstance -Query $BackupQuery

# --- 5. SQL Job Status (Last Run) ---
$JobQuery = @"
SELECT j.name AS JobName, 
    CASE jh.run_status WHEN 0 THEN 'Failed' WHEN 1 THEN 'Succeeded' WHEN 2 THEN 'Retry' WHEN 3 THEN 'Canceled' ELSE 'Unknown' END AS LastRunStatus,
    jh.run_date, jh.run_time
FROM msdb.dbo.sysjobs j
JOIN msdb.dbo.sysjobhistory jh ON j.job_id = jh.job_id
WHERE jh.instance_id = (SELECT MAX(instance_id) FROM msdb.dbo.sysjobhistory WHERE job_id = j.job_id)
"@
$JobStatus = Invoke-SqlCmd -ServerInstance $SQLInstance -Query $JobQuery

# --- Generate HTML Report ---
$Header = @"
<style>
    body { font-family: Calibri, sans-serif; }
    table { border-collapse: collapse; width: 100%; margin-bottom: 20px; }
    th { background-color: #4CAF50; color: white; padding: 8px; text-align: left; }
    td { border: 1px solid #ddd; padding: 8px; }
    tr:nth-child(even) { background-color: #f2f2f2; }
    .critical { color: red; font-weight: bold; }
</style>
"@

$Body = "<h2>SQL Server Health Report: $SQLInstance</h2>"
$Body += "<h3>Service Status</h3>" + ($Services | ConvertTo-Html -Fragment)
$Body += "<h3>Disk Space</h3>" + ($Disks | ConvertTo-Html -Fragment)
$Body += "<h3>Database Size & Status</h3>" + ($DBInfo | ConvertTo-Html -Fragment)
$Body += "<h3>Backup Status (Full)</h3>" + ($BackupStatus | ConvertTo-Html -Fragment)
$Body += "<h3>SQL Agent Job Status</h3>" + ($JobStatus | ConvertTo-Html -Fragment)

$FinalHTML = ConvertTo-Html -Head $Header -Body $Body

# --- Send Email ---
Send-MailMessage -To $EmailTo -From $EmailFrom -Subject "Daily Health Check: $SQLInstance" -Body ($FinalHTML | Out-String) -BodyAsHtml -SmtpServer $SMTPServer
