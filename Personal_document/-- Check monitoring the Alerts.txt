-- Check for active requests that might be slowing down monitoring tasks
SELECT 
    r.session_id, 
    r.status, 
    r.start_time, 
    r.total_elapsed_time / 1000 AS ElapsedTime_Sec,
    r.wait_type, 
    r.last_wait_type, 
    r.command,
    t.text AS QueryText
FROM sys.dm_exec_requests r
CROSS APPLY sys.dm_exec_sql_text(r.sql_handle) t
WHERE r.session_id > 50 -- Filter out system processes
ORDER BY r.total_elapsed_time DESC;
