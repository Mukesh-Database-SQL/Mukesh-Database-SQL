✅ 9. Check Backup Status
sql
Copy
Edit
SELECT 
    d.name AS [Database],
    MAX(b.backup_finish_date) AS [Last Backup],
    b.type AS [Type]
FROM 
    sys.databases d
LEFT JOIN 
    msdb.dbo.backupset b 
    ON d.name = b.database_name
GROUP BY 
    d.name, b.type
ORDER BY 
    [Last Backup];
Backup types:

D = Full

I = Differential

L = Log

✅ 10. Job Status (SQL Server Agent)
sql
Copy
Edit
SELECT 
    j.name AS [Job Name],
    ja.start_execution_date AS [Last Start],
    ja.stop_execution_date AS [Last End],
    ja.run_requested_date AS [Requested Start],
    h.run_status,
    h.message
FROM 
    msdb.dbo.sysjobs j
LEFT JOIN 
    msdb.dbo.sysjobactivity ja ON j.job_id = ja.job_id
LEFT JOIN 
    msdb.dbo.sysjobhistory h ON j.job_id = h.job_id AND h.step_id = 0
WHERE 
    ja.stop_execution_date IS NOT NULL;

✅ 12. CPU Usage and Memory
sql
Copy
Edit
-- CPU intensive queries
SELECT 
    TOP 10 
    total_worker_time/1000.0 AS [CPU (ms)],
    execution_count,
    total_worker_time / execution_count / 1000.0 AS [Avg CPU (ms)],
    SUBSTRING(qt.text, (qs.statement_start_offset/2)+1,
              ((CASE qs.statement_end_offset 
                  WHEN -1 THEN DATALENGTH(qt.text)
                  ELSE qs.statement_end_offset END 
                - qs.statement_start_offset)/2)+1) AS [Query Text]
FROM 
    sys.dm_exec_query_stats qs
CROSS APPLY 
    sys.dm_exec_sql_text(qs.sql_handle) AS qt
ORDER BY 
    total_worker_time DESC;
✅ 13. Index Fragmentation Report
sql
Copy
Edit
SELECT 
    dbschemas.[name] AS 'Schema',
    dbtables.[name] AS 'Table',
    dbindexes.[name] AS 'Index',
    indexstats.avg_fragmentation_in_percent
FROM 
    sys.dm_db_index_physical_stats (DB_ID(), NULL, NULL, NULL, 'LIMITED') AS indexstats
INNER JOIN 
    sys.tables dbtables ON dbtables.[object_id] = indexstats.[object_id]
INNER JOIN 
    sys.schemas dbschemas ON dbtables.[schema_id] = dbschemas.[schema_id]
INNER JOIN 
    sys.indexes AS dbindexes ON dbindexes.[object_id] = indexstats.[object_id]
    AND indexstats.index_id = dbindexes.index_id
WHERE 
    indexstats.avg_fragmentation_in_percent > 10
ORDER BY 
    indexstats.avg_fragmentation_in_percent DESC;
